    # 一个小的抽奖例子中，讲解代码优化

    从30人中，抽出三人中奖

    let perArr = new Array(30);
for (let index = 0; index < perArr.length; index++) {
  perArr[index] = index + "m";
}
for (let i = 0; i < 3; ++i) {
  let index = Math.floor(Math.random() * perArr.length);
  deleteRadomEle3(index);
}

// 方法一：去掉其中一个元素后，整体数组位置不发生改变,事件复杂度较高
function deleteRadomEle1(random) {
  return perArr.splice(random, 1);
}

// 方法四：原数组上操作，去掉其中一个元素后，整体数组去掉的元素和最后一位元素的位置发生互换，时间复杂度为O(2)
function deleteRadomEle4(random) {
  perArr[random] = perArr[perArr.length - 1];
  perArr.length--;
  return perArr;
}

// 方法二：生成新的数组，时间复杂度和空间复杂度都较高
function deleteRadomEle2(random) {
  return perArr =  perArr.slice(0, random).concat(perArr.slice(random + 1));
  
}

// 方法三：es6；生成新的数组，时间复杂度和空间复杂度都较高
function deleteRadomEle3(random) {
  return perArr =  [...perArr.slice(0, random), ...perArr.slice(random + 1)];
}
