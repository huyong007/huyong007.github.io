async function asyncFun(pramas){
    await const userName = promiseGetJson(params)
    .then(res=>res.userName);
    await const userInfo = promiseGetJson(userName[0])
    .then(res=>res.info);
    ...
    ...
}

这个是如何用生成器实现的呢？

先看下一个生成器函数的代码

function* generateFun(){
    yield '1';
    yield '2';
    yield '3';
}

const  iteratorObj = generateFun();

let item;
while (!(item = iteratorObj.next()).done) {
    console.log(item, item.value, 'iteratorObj value')
}

// { value: '1', done: false } 1 iteratorObj value
// { value: '2', done: false } 2 iteratorObj value
// { value: '3', done: false } 3 iteratorObj value

分析问题

  1. 实现生成器模拟async/await的关键是在第一个异步请求时（promiseGetJson），整个asyncFun能够等待第一个promise的finish或者是reject状态
  2. 拿到promiseGetJson的结果再进行第二个异步promiseGetJson获取userInfo
  3. 再来分析生成器，(generator函数特性1)当生成器generateFun函数执行后返回一个迭代器对象iteratorObj，我们使用迭代器对象来控制生成器的执行和挂起
  4. (generator函数特性2)iteratorObj迭代器对象暴露的next()接口可以向生成器请求值
  5. (generator函数特性3)请求的值为一个对象，对象有两个属性value和done;
  6. 只有在主动调用迭代器的next()方法时，生成器函数才能从挂起状态进入执行状态（前提是迭代器对象的done属性为false）
  7. **重点：**如果模拟async/await，我们就在第一次调用迭代器的next方法之后等待异步执行，等待proimse的异步执行完成之后将结果再传递给迭代器，进而使用该值进行第二次异步请求

function async(generator){
    // 步骤一.生成器函数从挂起状态开始，此时生成器内部的任何代码都未执行
    var iterator = generator();
    
    function handle(iteratorObj){
        if(iteratorObj.done){return;}
        // 步骤三.执行到yield语句终止，生成器转为挂起让渡状态
        // (*generator函数特性4*)此时第一个yield表达式的返回值是一个promise函数，➡️
        // ➡️promise也就是迭代器对象的value属性值
        const iteratorValue = iteratorObj.value;
        
        if(iteratorValue instanceof Promise){
            // 步骤四. promise的then方法中，激活生成器，生成器从让渡状态转为执行状态➡️
            // ➡️此时生成器还在第一个yield位置挂起等待下一次的调用
            // (*generator函数特性5*)此时传入next()方法的res就成为了第一个yield表达式的返回值，（特殊情况：如果想再第一个next()方法执行的时候就向生成器函数中传递至值直接在生成器函数的接受的变量处传值既可以），这样我们就完成了一次生成器内外的通信
            iteratorObj.then(res=>handle(iterator.next(res)))
            // 步骤五. iterator函数的throw可以用来抛出异常，终止生成器之后的执行
            .catch(err => iterator.throw(err));
        }
    }


    try{
        // 步骤二.激活生成器，从挂起状态转为执行状态，（如果生成器还存在值我们全局上下文中就存在一个对生成器函数的引用，这类似一个闭包，进而是生成器函数不会进入垃圾回收，每次生成器函数执行之后都是非阻塞的进入挂起让渡状态）
        handle(iterator.next());
    }
    catch(e){
        iteratorObj.throw(e);
        }
}

总结：

  • async/await使用了生成器函数的特性进行的封装，进而方便使用
  • 生成器函数执行➡️产生迭代器➡️迭代器通过next()控制生成器的挂起和执行并可以通过next()的参数和生成器交互
  • 迭代器通过next()方法返回的对象就是生成器执行到yield时候，yield表达式后面的值，该对象存在两个属性value和done
  • yield* 后面可以再加生成器函数，执行顺序为yield*后面的生成器函数执行之后，如果再调用next()再往下执行yield*所在的生成器
